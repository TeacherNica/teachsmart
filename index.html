function markAttendance(name,type){ const s=students.find(x=>x.name===name); if(!s)return; if(type==='present'){ if(s.classes<=0){alert(s.name+' has no classes left!');return;} s.classes--; s.attendance_today='present'; saveData();renderDashboard();renderStudents(); alert('✅ '+s.name+' present — 1 class deducted. '+s.classes+' left.'); }else{ s.attendance_today='absent'; saveData();renderDashboard(); alert('❌ '+s.name+' marked absent.'); } }
